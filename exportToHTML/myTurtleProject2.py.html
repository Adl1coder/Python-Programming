3    import turtle
4    import random
5    from time import sleep
6    
7    # Yıldız çizimi için turtle nesnesi oluşturulur.
8    star = turtle.Turtle()
9    
10   # Yıldızın kullanacağı renkleri içeren bir liste oluşturulur.
11   colours = ["black", "yellow", "blue", "purple"]
12   
13   # Yıldızın kalemi genişliği belirlenir.
14   star.width(3.5)
15   
16   # Yıldızın çizimine başlamadan önce kalemi kaldırır (penup).
17   star.penup()
18   
19   # Yıldızın başlangıç pozisyonuna gitmek için ileri doğru 20 birim hareket eder.
20   star.forward(20)
21   
22   # Yıldız deseninin çizildiği döngüye girilir.
23   for i in range(35):
24       # Yıldızın rengi rastgele bir renk seçer.
25       star.color(random.choice(colours))
26   
27       # Yıldızın bir sonraki çizgi uzunluğunu artırarak yıldızı çizer.
28       star.forward(i * 10)
29       star.right(144)
30   
31   # Yıldızın rengi gri yapılır.
32   star.color('gray')
33   
34   # mesaj ekrana yazdırılır.
35   message2 = "\t\t\t\t\t Güzel mi?   Teşekkür ederim :) "
36   star.write(message2, font=("Arial", 13, "normal"))
37   print(message2)
38   
39   # 3 saniye beklenir.
40   sleep(3)
41   
42   # Yıldızın üzerini temizler ve "Bekle..." mesajı ekrana yazdırılır.
43   star.clear()
44   message = "\t\t\t\t\t\t\t\t Bekle..."
45   star.write(message, font=("Arial", 30, "italic"))
46   print(message)
47   
48   # 3 saniye beklenir.
49   sleep(3)
50   
51   # Yıldızın başlangıç pozisyonuna dönmesi için ev işareti çizer.
52   star.clear()
53   star.home()
54   star.penup()
55   star.forward(20)
56   star.pendown()
57   
58   # Yıldız deseninin farklı bir renkte tekrar çizilmesi.
59   for i in range(35):
60       star.color(random.choice(colours))
61       star.forward(i * 10)
62       star.right(144)
63   
64   # 5 saniye beklenir.
65   sleep(5)
66   
67   # Yıldızın üzerini temizler ve "Thanks for watching!" mesajı ekrana yazdırılır.
68   star.clear()
69   star.color('blue')
70   message = "\t\t\t\t\t\t\t İzlemene sevindim!"
71   star.write(message, font=("Arial", 20, "normal"))
72   print(message)
73   
74   # Turtle penceresini kapatır ve programı tamamlar.
75   turtle.done()
